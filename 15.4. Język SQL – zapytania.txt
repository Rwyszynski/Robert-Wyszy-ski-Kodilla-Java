SELECT S.SUMMARY, I.NAME
FROM ISSUES S
JOIN ISSUESLISTS I ON S.ISSUESLIST_ID = I.ID;

SELECT S.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES S
JOIN USERS U ON S.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*)
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME
HAVING COUNT(*) > 0 ;
